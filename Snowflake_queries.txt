3)LIST OF TABLES IN SNOWFLAKE SCHEMA GROUPED BY DATABASE AND SCHEMA

SELECT TABLE_CATALOG AS DATABASE_NAME, TABLE_SCHEMA AS SCHEMA_NAME, TABLE_NAME
FROM INFORMATION_SCHEMA.TABLES
WHERE TABLE_TYPE = 'BASE TABLE'
ORDER BY DATABASE_NAME, SCHEMA_NAME, TABLE_NAME;

4) RETRIVE THE LIST OF VIEW IN SNOWFLAKE.

SELECT TABLE_CATALOG AS DATABASE_NAME, TABLE_SCHEMA AS SCHEMA_NAME, TABLE_NAME
FROM INFORMATION_SCHEMA.TABLES
WHERE TABLE_TYPE = 'VIEW'
ORDER BY DATABASE_NAME, SCHEMA_NAME, TABLE_NAME;

5) HOW TO SEE THE LIST OF task and stream details in snowflake

6) How to see the list of stored procedures

SELECT ROUTINE_CATALOG AS DATABASE_NAME, ROUTINE_SCHEMA AS SCHEMA_NAME, ROUTINE_NAME
FROM INFORMATION_SCHEMA.ROUTINES
WHERE ROUTINE_TYPE = 'PROCEDURE'
ORDER BY DATABASE_NAME, SCHEMA_NAME, ROUTINE_NAME;


7) how to see the list of fucntions

SELECT ROUTINE_CATALOG AS DATABASE_NAME, ROUTINE_SCHEMA AS SCHEMA_NAME, ROUTINE_NAME
FROM INFORMATION_SCHEMA.ROUTINES
WHERE ROUTINE_TYPE = 'FUNCTION'
ORDER BY DATABASE_NAME, SCHEMA_NAME, ROUTINE_NAME;
